#include <DHT11.h>
int temp= 0;
int umid=0;

DHT11 sensor(9);

void setup() {
  pinMode(7,INPUT);
  Serial.begin(9600);
}

void loop() {
  int resultado = sensor.readTemperatureHumidity(temp,umid);
  Serial.print("Temperatura: ");
  Serial.print(temp);
  Serial.print("Umidade: %");
  Serial.println(umid);
  delay(500);
}
